export default async function handler(req, res) { const { prompt } = req.body; const response = await fetch("https://api.openai.com/v1/responses", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": "Bearer " + process.env.OPENAI_API_KEY }, body: JSON.stringify({ model: "gpt-5.3", input: prompt }) }); const data = await response.json(); res.status(200).json({ text: data.output[0].content[0].text }); }
